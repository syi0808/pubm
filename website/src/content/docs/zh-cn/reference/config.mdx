本页描述 `@pubm/core` 的 `defineConfig()` 暴露的 `pubm.config.*` 类型结构。

当前 publish 流程最关键的字段有：

- `packages`
- `plugins`
- `versioning`、`fixed`、`linked`
- `snapshot.useCalculatedVersion`

## 默认配置

```ts
export default defineConfig({
  versioning: "independent",
  branch: "main",
  changelog: true,
  changelogFormat: "default",
  commit: false,
  access: "public",
  fixed: [],
  linked: [],
  updateInternalDependencies: "patch",
  ignore: [],
  tag: "latest",
  contents: ".",
  saveToken: true,
  releaseDraft: true,
  releaseNotes: true,
  rollbackStrategy: "individual",
  packages: [{ path: "." }],
  validate: {
    cleanInstall: true,
    entryPoints: true,
    extraneousFiles: true,
  },
  snapshot: {
    useCalculatedVersion: false,
    prereleaseTemplate: "{tag}-{timestamp}",
  },
});
```

## 主要字段

### `versioning`

- 类型: `"independent" | "fixed"`
- 默认: `"independent"`

### `branch`

- 类型: `string`
- 默认: `"main"`

### `packages`

- 类型: `PackageConfig[]`
- 默认: `[{ path: "." }]`

registry 从清单文件中自动推断，可通过 `packages[].registries` 覆盖。

```ts
packages: [
  { path: "packages/core", ecosystem: "js" },
  { path: "crates/pubm-rs", ecosystem: "rust" },
];
```

添加私有 registry：

```ts
packages: [
  {
    path: "packages/a",
    registries: [
      "npm",
      { url: "https://npm.internal.com", token: { envVar: "INTERNAL_NPM_TOKEN" } },
    ],
  },
];
```

### `fixed`

- 类型: `string[][]`
- 默认: `[]`

### `linked`

- 类型: `string[][]`
- 默认: `[]`

### `tag`

- 类型: `string`
- 默认: `"latest"`

### `contents`

- 类型: `string`
- 默认: `"."`

### `saveToken`

- 类型: `boolean`
- 默认: `true`

### `releaseDraft`

- 类型: `boolean`
- 默认: `true`

### `plugins`

- 类型: `PubmPlugin[]`
- 默认: `[]`

### `validate`

```ts
{
  cleanInstall: true,
  entryPoints: true,
  extraneousFiles: true,
}
```

### `snapshot`

```ts
{
  useCalculatedVersion: false,
  prereleaseTemplate: "{tag}-{timestamp}",
}
```
